# O(m + n) time

class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """
        # 用三个指针，其中两个指针用来比大小，另一个指针用来标记放置的位置

        nums1_i = m - 1
        nums2_i = n - 1
        merge_i = m + n - 1

        while nums2_i >= 0 :
            if nums1_i < 0:
                nums1[merge_i] = nums2[nums2_i]
                nums2_i -=1 
            elif nums2[nums2_i] >= nums1[nums1_i]:
                nums1[merge_i] = nums2[nums2_i]
                nums2_i -= 1
            else:
                nums1[merge_i] = nums1[nums1_i]
                nums1_i -= 1
            
            merge_i -= 1
