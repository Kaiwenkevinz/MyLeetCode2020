### 桶
需要O(n) 空间，题目要求常量空间，WE CAN DO BETTER!
```py
# Use dictionary O(n) space
class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Use bucket to get the frequency
        bucket = [0] * 3
        for n in nums:
            bucket[n] += 1

        # Modify nums in place
        n = 0
        i = 0
        while n <= 2:
            freq = bucket[n]
            count = 0
            while count < freq:
                nums[i] = n
                i += 1
                count += 1
            n += 1

```

### 单指针，扫描两次
第一次扫描把0移到最左边，第二次扫描把1移到最后一个0的左边，剩下的就是2了  
O(n)时间，O(1)空间
```py
class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        # single pointer, scan twice
        # first scan move 0 to the left
        # second scan move 1 to the left

        pt = 0

        for i in range(len(nums)):
            if nums[i] == 0:
                self.swap(nums, pt, i)
                pt += 1
        
        for i in range(len(nums)):
            if nums[i] == 1:
                self.swap(nums, pt, i)
                pt += 1
        
    
    def swap(self, nums, i, j):
        temp = nums[i]
        nums[i] = nums[j]
        nums[j] = temp
```

### 双指针，扫描一次
快慢指针
```py
class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        pt_zero = 0
        pt_one = 0

        for i in range(len(nums)):
            if nums[i] == 1:
                self.swap(nums, i, pt_one)
                pt_one += 1
            elif nums[i] == 0:
                self.swap(nums, i, pt_zero)
                if pt_zero < pt_one:
                    self.swap(nums, i, pt_one)
                pt_zero += 1
                pt_one += 1
    
    def swap(self, nums, i, j):
        temp = nums[i]
        nums[i] = nums[j]
        nums[j] = temp
```
