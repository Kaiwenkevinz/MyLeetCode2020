### 排序
使用库排序，直接找到kth元素  
python sort()使用了Timsort, 是mergesort和insertionsort的结合体, 具有稳定的特性, O(nlogn) time, O(n) space
```py
class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        nums.sort()
        return nums[-k]
```

### Partition
```py
class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        tar_i = len(nums) - k
        right = len(nums) - 1
        left = 0

        while 1:
            index = self.partition(nums, left, right)
            if index == tar_i:
                return nums[index]
            elif index < tar_i:
                left = index + 1
            else:
                right = index - 1
                
    def swap(self, nums, a, b):
        temp = nums[a]
        nums[a] = nums[b]
        nums[b] = temp

    # https://www.geeksforgeeks.org/quick-sort/
    def partition(self, nums, left, right):
        pivot = nums[right]
        i = left - 1
        j = left
        while j < right:
            if nums[j] < pivot:
                i += 1
                print(i)
                self.swap(nums, i, j)
            else:
                j -= 1
        self.swap(nums, i + 1, right)

        return i + 1
```

### Heap
```py
from queue import PriorityQueue
class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        q = PriorityQueue()
        for e in nums:
            q.put(e) # O(logn)

        res = 0
        for i in range(len(nums) - k + 1):
            res = q.get()   # O(logn)

        return res
```
