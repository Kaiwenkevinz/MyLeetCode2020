递归造树，不改变t1和t2  
时间花费O(n),空间花费O(h)
```py
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def mergeTrees(self, t1: TreeNode, t2: TreeNode) -> TreeNode:
        if not t1 and not t2:
            return None

        newR = None

        if t1 and t2:
            newR = TreeNode(t1.val + t2.val)
            newR.left = self.mergeTrees(t1.left, t2.left)
            newR.right = self.mergeTrees(t1.right, t2.right)
        elif t1 and not t2:
            newR = TreeNode(t1.val)
            newR.left = self.mergeTrees(t1.left, None)
            newR.right = self.mergeTrees(t1.right, None)
        elif t2 and not t1:
            newR = TreeNode(t2.val)
            newR.left = self.mergeTrees(None, t2.left)
            newR.right = self.mergeTrees(None, t2.right)

        return newR
```

直接把t2合并到t1上，速度更快但是修改了函数参数t1
```py
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def mergeTrees(self, t1: TreeNode, t2: TreeNode) -> TreeNode:
        if not t1 and not t2:
            return None

        if t1 and t2:
            t1.val += t2.val
            t1.left = self.mergeTrees(t1.left, t2.left)
            t1.right = self.mergeTrees(t1.right, t2.right)
        else:
            if not t1:
                return t2
            if not t2:
                return t1        

        return t1
```
