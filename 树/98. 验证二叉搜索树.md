中序遍历将BST转换为数组，然后检查数组是否为升序  
空间花费较大，为O(n),考虑优化
```py
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isValidBST(self, root: TreeNode) -> bool:

        res = []
        self.helper(root, res)

        for i in range(len(res) - 1):
            if res[i] >= res[i + 1]:
                return False
        
        return True

    
    def helper(self, root, res):
        if not root:
            return

        self.helper(root.left, res)
        res.append(root.val)
        self.helper(root.right, res)
```

### 递归+全局变量
如果不算递归栈的话，空间花费优化为 O(1)
```py
class Solution:
    preVal = -float("inf")
    res = True

    def isValidBST(self, root: TreeNode) -> bool:

        if not root:
            return
        
        self.isValidBST(root.left)

        if root.val <= self.preVal:
            self.res = False
            return self.res
        
        self.preVal = root.val

        self.isValidBST(root.right)

        return self.res
```

### 迭代
中序遍历的迭代实现
```py
class Solution:
    def isValidBST(self, root: TreeNode) -> bool:

        preVal = -float("inf")
        s = []
        while s or root:
            while root:
                s.append(root)
                root = root.left
            
            root = s.pop()
            if root.val <= preVal:
                return False
            preVal = root.val
            root = root.right
        
        return True
```
