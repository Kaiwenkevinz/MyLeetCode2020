### BFS迭代

```py
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def levelOrder(self, root: TreeNode) -> List[List[int]]:

        if not root:
            return []

        res =[]

        q = [root]
        while q:
            curSize = len(q)
            curNodes = []
            for i in range(curSize):
                node = q.pop(0)
                curNodes.append(node.val)
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            res.append(curNodes)
            
        return res
```
