class Solution:
    def levelOrderBottom(self, root: TreeNode) -> List[List[int]]:
        
        if not root:
            return []
            
        res = []
        s = [root]

        while s:
            curSize = len(s)
            temp = []
            for _ in range(curSize):
                n = s.pop(0)
                temp.append(n.val)
                if n.left:
                    s.append(n.left)
                if n.right:
                    s.append(n.right)
            res.append(temp)        

        return res[::-1]
