用队列实现从右到左的层级遍历，每层只append第一个node
```py
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def rightSideView(self, root: TreeNode) -> List[int]:
        if not root:
            return []

        res = []
        q = [root]
        
        while q:
            added = False
            curSize = len(q)
            for _ in range(curSize):
                n = q.pop(0)
                if not added:
                    res.append(n.val)
                    added = True
                if n.right:
                    q.append(n.right)
                if n.left:
                    q.append(n.left)

        return res
```
