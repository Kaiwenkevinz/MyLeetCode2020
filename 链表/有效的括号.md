class Solution {
    public boolean isValid(String s) {
        Map<Character, Character> map = new HashMap<>();
        map.put(')', '(');
        map.put(']', '[');
        map.put('}', '{');
        Stack<Character> stack = new Stack<>();
        for (int i = 0; i < s.length(); i++) {
            Character c = s.charAt(i);
            if (map.containsKey(c)) {
                if (stack.empty()) {
                    return false;
                }
                if (!stack.peek().equals(map.get(c))) {
                    return false;
                }
                stack.pop();
            } else {
                stack.add(c);
            }
        }
        if (!stack.empty()) {
            return false;
        }
        return true;
    }
}
