public class Solution {
    public ListNode Merge(ListNode list1,ListNode list2) {
        if(list1 == null && list2 == null) {
            return null;
        }
        
        ListNode node = new ListNode(1);
        ListNode cur = node;
        while(list1 != null && list2 != null) {
            if(list1.val < list2.val) {
                cur.next = list1;
                list1 = list1.next;
            } else {
                cur.next = list2;
                list2 = list2.next;
            }
            cur = cur.next;
        }
        
        while(list1 != null) {
            cur.next = list1;
            cur = cur.next;
            list1 = list1.next;
        }
        
        while(list2 != null) {
            cur.next = list2;
            cur = cur.next;
            list2 = list2.next;
        }
        
        return node.next;
    }
}
