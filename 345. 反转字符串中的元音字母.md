# time O(n)
# space O(1)

class Solution:
    def reverseVowels(self, s: str) -> str:
        letters = ['a','e','i','o','u', 'A', 'E', 'I', 'O', 'U']
        list_s = list(s)
        l = 0
        r = len(list_s) - 1

        while l < r:
            c_l = list_s[l]
            c_r = list_s[r]

            if c_l in letters and c_r in letters:
                list_s[l] = c_r
                list_s[r] = c_l
                l += 1
                r -= 1
            else:
                if c_l not in letters:
                    l += 1
                if c_r not in letters:
                    r -= 1
        
        return "".join(list_s)
